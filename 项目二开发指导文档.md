# 项目二：AI驱动的数据处理平台 — 开发指导文档

> [!abstract] 项目定位
> Django + Celery + LangChain 构建的数据处理平台。用户上传 CSV/Excel，用自然语言查数据，AI 自动生成 SQL 并执行返回结果。核心展示 **后端工程化能力**：RBAC 权限、幂等上传、Celery 任务可靠性、批量写入优化。

## 一、技术栈

| 层级 | 技术 | 用途 | 选型理由 |
|------|------|------|----------|
| Web 框架 | Django + DRF | RESTful API + Admin 后台 | Admin 开箱即用，ORM 原生支持 PostgreSQL 高级功能 |
| 关系数据库 | PostgreSQL | 业务数据 + NL2SQL 查询执行 | Django 官方推荐，窗口函数/CTE 丰富，LLM 训练数据中 PG-SQL 示例多 |
| 异步任务 | Celery + Redis | 大文件异步处理 + 任务可靠性 | CPU 密集型运算不适合 async，Celery 有重试/死信队列/横向扩展能力 |
| 数据处理 | Pandas | CSV/Excel 解析与统计分析 | 数据清洗比 SQL 灵活，和 Python 生态无缝集成 |
| AI 能力 | LangChain | NL2SQL 自然语言查询 | text-to-sql 生态成熟，Agent 编排灵活 |
| 前端 | Vue3 + Element Plus | 组件化查询界面 | el-table/el-pagination 开箱即用，比手写 Bootstrap 更快 |
| 部署 | Docker + Docker Compose | 容器化部署 | 一键启动完整服务栈 |

## 二、技术架构

```
用户上传文件 → DRF API
                │
                ├── 幂等检查（MD5哈希 + Redis分布式锁）
                │
                ├── 小文件（<10MB）→ 同步Pandas处理 → PostgreSQL
                │
                └── 大文件（≥10MB）→ Celery异步任务
                                       │
                                       ├── Pandas分块读取（内存控制200MB内）
                                       ├── 数据清洗/类型推断
                                       ├── 写入PostgreSQL（bulk_create批量插入）
                                       ├── WebSocket通知完成
                                       └── 失败 → 指数退避重试 → 死信队列

自然语言查询 → RBAC权限校验 → LangChain NL2SQL
                 │
                 ├── 查询意图解析
                 ├── SQL生成（基于表结构Schema）
                 ├── 安全审查（禁止DROP/DELETE等）
                 └── 执行SQL → 结果格式化返回
```

```python
from django.db import models


class User(AbstractUser):

        db_table = 'users'

    def is_admin(self):

    @property
    def is_analyst(self):
```

```python
# apps/users/permissions.py
from rest_framework.permissions import BasePermission


class IsAdmin(BasePermission):
    """管理员权限：全权限 + 用户管理"""
    message = '需要管理员权限'

    def has_permission(self, request, view):
        return request.user.is_authenticated and request.user.role == 'admin'


class IsAnalyst(BasePermission):
    """分析师权限：上传 + 查询 + 导出"""
    message = '需要分析师权限'

    def has_permission(self, request, view):
        return request.user.is_authenticated and request.user.role in (
            'admin', 'analyst'
        )


class IsViewer(BasePermission):
    """只读权限：仅查询"""
    message = '需要至少只读权限'

    def has_permission(self, request, view):
        return request.user.is_authenticated and request.user.role in (
            'admin', 'analyst', 'viewer'
        )
```

### 幂等上传服务（MD5 + Redis 分布式锁）

```python
# apps/datasets/services/upload_lock.py
import redis
from django.conf import settings


class UploadLockService:
    """文件上传幂等性服务

    通过 MD5 哈希去重 + Redis SETNX 分布式锁防止并发重复处理。
    双重保障：分布式锁是第一道防线，数据库 file_md5 唯一约束是第二道防线。
    """

    def __init__(self):
        self.redis_client = redis.from_url(settings.REDIS_URL)
        self.lock_timeout = 300  # 5分钟过期

    def try_acquire_lock(self, file_md5: str) -> bool:
        """尝试获取分布式锁

        SETNX：key不存在时设置成功返回True，已存在返回False
        + 过期时间：防止死锁（处理进程崩溃后锁自动释放）
        """
        lock_key = f'upload_lock:{file_md5}'
        acquired = self.redis_client.set(lock_key, '1', nx=True, ex=self.lock_timeout)
        return bool(acquired)

    def release_lock(self, file_md5: str):
        """手动释放锁"""
        lock_key = f'upload_lock:{file_md5}'
        self.redis_client.delete(lock_key)

    def is_file_processed(self, file_md5: str) -> bool:
        """检查文件是否已被处理（查数据库唯一约束）"""
        from apps.datasets.models import Dataset
        return Dataset.objects.filter(file_md5=file_md5).exists()
```

### 上传接口

```python
# apps/datasets/api/views.py
import hashlib
from rest_framework import viewsets, status
from rest_framework.response import Response
from rest_framework.parsers import MultiPartParser

from apps.users.permissions import IsAnalyst, IsViewer
from apps.datasets.models import Dataset
from apps.datasets.serializers import DatasetSerializer
from apps.datasets.services.upload_lock import UploadLockService


class DatasetViewSet(viewsets.ModelViewSet):
    """数据集管理接口"""

    serializer_class = DatasetSerializer
    parser_classes = [MultiPartParser]

    def get_permissions(self):
        """根据action动态设置权限"""
        if self.action in ('create', 'update', 'destroy'):
            permission_classes = [IsAnalyst]
        else:
            permission_classes = [IsViewer]
        return [perm() for perm in permission_classes]

    def get_queryset(self):
        """行级数据隔离：用户只能看到自己的数据集"""
        return Dataset.objects.filter(owner=self.request.user)

    def create(self, request, *args, **kwargs):
        """上传数据集（幂等设计）"""
        file = request.FILES.get('file')
        if not file:
            return Response(
                {'error': '请选择文件'}, status=status.HTTP_400_BAD_REQUEST
            )

        # 1. 计算文件MD5
        md5_hash = hashlib.md5()
        for chunk in file.chunks():
            md5_hash.update(chunk)
        file_md5 = md5_hash.hexdigest()

        # 2. 幂等检查：是否已处理过（数据库唯一约束）
        lock_service = UploadLockService()
        if lock_service.is_file_processed(file_md5):
            return Response(
                {'error': '该文件已存在', 'file_md5': file_md5},
                status=status.HTTP_409_CONFLICT,
            )

        # 3. 获取分布式锁（Redis SETNX）
        if not lock_service.try_acquire_lock(file_md5):
            return Response(
                {'error': '该文件正在处理中'},
                status=status.HTTP_409_CONFLICT,
            )

        try:
            # 4. 创建数据集记录
            dataset = Dataset.objects.create(
                name=request.data.get('name', file.name),
                description=request.data.get('description', ''),
                file_name=file.name,
                file_size=file.size,
                file_md5=file_md5,
                owner=request.user,
            )

            # 5. 判断文件大小决定同步/异步处理
            if file.size < 10 * 1024 * 1024:  # <10MB 同步处理
                from apps.datasets.services.parser import parse_file_sync
                parse_file_sync(dataset, file)
            else:  # >=10MB 异步处理
                from apps.datasets.tasks import process_large_file
                process_large_file.delay(str(dataset.id))

            serializer = DatasetSerializer(dataset)
            return Response(serializer.data, status=status.HTTP_201_CREATED)

        except Exception as e:
            lock_service.release_lock(file_md5)
            dataset.status = 'failed'
            dataset.save()
            return Response(
                {'error': str(e)}, status=status.HTTP_500_INTERNAL_SERVER_ERROR
            )
```

```python
        dataset.save()
        raise e


def clean_dataframe(df: pd.DataFrame) -> pd.DataFrame:
    """数据清洗管道"""
    # 清理列名（去空格、统一小写、空格转下划线）
    df.columns = df.columns.str.strip().str.lower().str.replace(' ', '_')

    # 自动类型推断
    for col in df.columns:
        df[col] = pd.to_numeric(df[col], errors='ignore')

    # 处理缺失值（NaN → None，JSONField兼容）
    df = df.where(pd.notnull(df), None)

    return df


def bulk_create_rows(dataset: Dataset, df: pd.DataFrame, batch_size: int = 1000):
    """批量写入数据行

    对比逐条save：
    - 逐条save：每条一次INSERT SQL + 触发signal，1000条 = 1000次DB往返
    - bulk_create：一条INSERT多条值，1000条 = 1次DB往返
    - 实测1万条：逐条save约30秒，bulk_create约0.3秒，100倍差距

    注意：bulk_create不触发signal，如果逻辑依赖post_save信号则不适合
    """
    rows = []
    for index, row in df.iterrows():
        rows.append(
            DataRow(
                dataset=dataset,
                row_index=index,
                data=row.to_dict(),
            )
        )
        if len(rows) >= batch_size:
            DataRow.objects.bulk_create(rows)
            rows = []

    if rows:
        DataRow.objects.bulk_create(rows)
```

## 八、模块四：Celery 异步任务 + 重试 + 死信队列

### 异步任务

```python
# apps/datasets/tasks.py
import pandas as pd
from celery import shared_task
from celery.utils.log import get_task_logger

logger = get_task_logger(__name__)


@shared_task(
    bind=True,
    max_retries=5,
    default_retry_delay=0,  # 使用自定义退避
    acks_late=True,  # 任务完成后才确认，防止中途丢失
)
def process_large_file(self, dataset_id: str):
    """异步处理大文件

    重试策略：指数退避 2^n 秒（1→2→4→8→16），最多5次
    超过5次标记为failed（死信），Admin后台可见

    为什么不固定间隔重试？
    固定间隔1秒重试，如果失败原因是下游暂时不可用，1秒可能还没恢复。
    指数退避给下游恢复的时间越来越长，成功率更高。
    """
    from apps.datasets.models import Dataset, DataRow
    from apps.datasets.services.parser import clean_dataframe

    try:
        dataset = Dataset.objects.get(id=dataset_id)
        dataset.status = 'processing'
        dataset.save()

        # Pandas分块读取，控制内存峰值在200MB内
        chunk_size = 10000
        total_rows = 0
        file_path = dataset.file.path

        if dataset.file_name.endswith('.csv'):
            reader = pd.read_csv(file_path, chunksize=chunk_size)
        else:
            reader = pd.read_excel(file_path, chunksize=chunk_size)

        for chunk_df in reader:
            chunk_df = clean_dataframe(chunk_df)

            rows = [
                DataRow(dataset=dataset, row_index=total_rows + i, data=row.to_dict())
                for i, row in chunk_df.iterrows()
            ]
            DataRow.objects.bulk_create(rows, batch_size=1000)

            total_rows += len(chunk_df)

            # WebSocket进度推送（见模块五）
            progress = min(int(total_rows / (dataset.row_count or 1) * 100), 99)
            logger.info(f'Dataset {dataset_id}: {progress}% ({total_rows} rows)')

        dataset.row_count = total_rows
        dataset.status = 'completed'
        dataset.save()

    except Exception as exc:
        logger.error(f'处理数据集 {dataset_id} 失败: {exc}')

        dataset = Dataset.objects.get(id=dataset_id)

        if self.request.retries < self.max_retries:
            countdown = 2 ** self.request.retries
            logger.info(f'重试 {self.request.retries + 1}/{self.max_retries}，等待 {countdown}秒')
            raise self.retry(exc=exc, countdown=countdown)
        else:
            dataset.status = 'failed'
            dataset.save()
            logger.error(f'数据集 {dataset_id} 重试{self.max_retries}次后仍然失败，进入死信队列')
```

### NL2SQL 服务

```python
# apps/query/services/nl2sql.py
from langchain_openai import ChatOpenAI
from langchain.chains import create_sql_query_chain
from langchain_community.utilities import SQLDatabase
from django.conf import settings


class NL2SQLService:
    """自然语言转SQL服务"""

    # SQL安全审查：只允许SELECT
    DANGEROUS_KEYWORDS = [
        'DROP', 'DELETE', 'UPDATE', 'INSERT', 'ALTER',
        'CREATE', 'TRUNCATE', 'EXEC', 'EXECUTE',
        'GRANT', 'REVOKE',
    ]

    def __init__(self, dataset):
        self.dataset = dataset
        self.db = self._get_db_connection()
        self.llm = ChatOpenAI(
            model='gpt-3.5-turbo',
            temperature=0,  # SQL生成用0温度保证稳定性
            openai_api_key=settings.OPENAI_API_KEY,
        )

    def _get_db_connection(self):
        """创建SQLAlchemy连接（指向PostgreSQL）"""
        db_settings = settings.DATABASES['default']
        db_url = (
            f"postgresql://{db_settings['USER']}:{db_settings['PASSWORD']}"
            f"@{db_settings['HOST']}:{db_settings['PORT']}/{db_settings['NAME']}"
        )
        return SQLDatabase.from_uri(db_url)

    def _validate_sql(self, sql: str) -> bool:
        """SQL安全审查（三层防护）

        1. 白名单机制：只允许SELECT语句
        2. 禁止危险关键字（DROP/DELETE/UPDATE等）
        3. 禁止多语句（分号注入）
        """
        sql_stripped = sql.strip().upper()

        if not sql_stripped.startswith('SELECT'):
            return False

        for keyword in self.DANGEROUS_KEYWORDS:
            if keyword in sql_stripped:
                return False

        # 禁止多语句（最后一个分号是允许的）
        if ';' in sql_stripped[:-1]:
            return False

        return True

    def query(self, question: str) -> dict:
        """自然语言查询数据

        Args:
            question: 自然语言问题，如"上个月销售额最高的10个产品"

        Returns:
            {'sql': 生成的SQL, 'results': 查询结果, 'success': bool}
        """
        # 1. 生成SQL
        chain = create_sql_query_chain(self.llm, self.db)

        prompt_context = {
            'input': question,
            'table_info': self.db.get_table_info(
                table_names=['dataset_rows']
            ),
            'dialect': 'postgresql',
        }

        sql = chain.invoke(prompt_context)

        # 2. 安全审查
        if not self._validate_sql(sql):
            return {
                'error': '生成的SQL包含不允许的操作，仅支持SELECT查询',
                'sql': sql,
                'success': False,
            }

        # 3. 自动加LIMIT防止返回过多数据
        if 'LIMIT' not in sql.upper():
            sql = sql.rstrip(';') + ' LIMIT 1000;'

        # 4. 执行SQL
        try:
            results = self.db.run(sql)
            return {
                'sql': sql,
                'results': results,
                'success': True,
            }
        except Exception as e:
            return {
                'sql': sql,
                'error': str(e),
                'success': False,
            }
```

```python
# apps/query/api/views.py
from rest_framework.views import APIView
from rest_framework.response import Response
from rest_framework import status

from apps.users.permissions import IsViewer
from apps.datasets.models import Dataset
from apps.query.services.nl2sql import NL2SQLService


class NL2SQLQueryView(APIView):
    """自然语言查询接口"""
    permission_classes = [IsViewer]

    def post(self, request, dataset_id):
        try:
            dataset = Dataset.objects.get(
                id=dataset_id,
                owner=request.user,  # 行级隔离
            )
        except Dataset.DoesNotExist:
            return Response(
                {'error': '数据集不存在'}, status=status.HTTP_404_NOT_FOUND
            )

        if dataset.status != 'completed':
            return Response(
                {'error': '数据集尚未处理完成'}, status=status.HTTP_400_BAD_REQUEST
            )

        question = request.data.get('question', '')
        if not question:
            return Response(
                {'error': '请输入查询问题'}, status=status.HTTP_400_BAD_REQUEST
            )

        service = NL2SQLService(dataset)
        result = service.query(question)

        # 记录查询历史
        from apps.query.models import QueryHistory
        QueryHistory.objects.create(
            dataset=dataset,
            user=request.user,
            question=question,
            generated_sql=result.get('sql', ''),
            is_success=result.get('success', False),
        )

        return Response(result)
```

```python
import io
import pandas as pd


class ExportService:
    """数据导出服务"""

    @staticmethod
    def export_csv(results: list[dict]) -> str:
        """导出查询结果为CSV"""
        df = pd.DataFrame(results)
        return df.to_csv(index=False)

    @staticmethod
    def export_excel(results: list[dict]) -> bytes:
        """导出查询结果为Excel"""
        df = pd.DataFrame(results)
        buffer = io.BytesIO()
        with pd.ExcelWriter(buffer, engine='openpyxl') as writer:
            df.to_excel(writer, index=False, sheet_name='查询结果')
        return buffer.getvalue()
```
